174def merge_config(base: Dict[str, Any], override: Dict[str, Any]) -> Dict[str, Any]:
175    merged = base.copy()
176    for k, v in override.items():
177        if v is not None:
178            merged[k] = v

182def resolve_final_config(
183    preset_name: Optional[str],
184    config_file: Optional[str],
185    cli_overrides: Dict[str, Any],
186) -> Dict[str, Any]:
187    # 1. Load preset or custom config
188    if config_file:
189        base = load_custom_config(config_file)
190    elif preset_name:
191        base = load_config_preset(preset_name)
192    else:
193        base = {}
194    # 2. Merge CLI overrides
195    final = merge_config(base, cli_overrides)
196    return final
