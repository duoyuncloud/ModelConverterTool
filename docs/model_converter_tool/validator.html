 25        output_format: str,

 27        quantization_type: str = None,

 29        """

 72    def _validate_onnx_model(self, model_path: str, model_type: str) -> Dict[str, Any]:
 73        """Validate ONNX model with detailed error reporting"""
 74        try:
 75            import traceback
 76
 77            import onnxruntime as ort
 78
 79            # 优先直接检测 model_path 是否为 .onnx 文件
 80            model_path = Path(model_path)
 81            if model_path.is_file() and model_path.suffix == ".onnx":
 82                onnx_file = model_path
 83            else:
 84                # Find ONNX file in directory
 85                onnx_files = list(model_path.glob("*.onnx"))
 86                if not onnx_files:
 87                    return {"success": False, "error": "No ONNX files found"}
 88                onnx_file = onnx_files[0]
 89
 90            # Load ONNX model
 91            try:
 92                session = ort.InferenceSession(str(onnx_file))
 93            except Exception as e:
 94                return {
 95                    "success": False,
 96                    "error": f"ONNX load failed: {e}",
 97                    "traceback": traceback.format_exc(),
 98                }
 99
100            # Get input/output info
101            try:
102                input_name = session.get_inputs()[0].name
103                output_name = session.get_outputs()[0].name
104            except Exception as e:
105                return {
106                    "success": False,
107                    "error": f"ONNX input/output info extraction failed: {e}",
108                    "traceback": traceback.format_exc(),
109                }
110
111            # Create test input based on model type
112            try:
113                input_names = [inp.name for inp in session.get_inputs()]
114                input_shapes = [inp.shape for inp in session.get_inputs()]
115                input_types = [inp.type for inp in session.get_inputs()]
116                input_feed = {}
117                for name, shape, typ in zip(input_names, input_shapes, input_types):
118                    shape = tuple(1 if isinstance(x, str) or x is None else x for x in shape)
119                    if "int" in typ:
120                        arr = np.ones(shape, dtype=np.int64)
121                    else:
122                        arr = np.ones(shape, dtype=np.float32)
123                    input_feed[name] = arr
124            except Exception as e:
125                return {
126                    "success": False,
127                    "error": f"ONNX dummy input creation failed: {e}",
128                    "traceback": traceback.format_exc(),
129                }
130
131            # Run inference
132            try:
133                outputs = session.run([output_name], input_feed)
134            except Exception as e:
135                return {
136                    "success": False,
137                    "error": f"ONNX inference failed: {e}",
138                    "traceback": traceback.format_exc(),
139                }
140
141            # Validate output
142            if len(outputs) == 0:
143                return {"success": False, "error": "No outputs from ONNX model"}
144
145            output = outputs[0]
146
147            return {
148                "success": True,
149                "model_file": str(onnx_file),
150                "input_shape": input_feed[input_name].shape,
151                "output_shape": output.shape,
152                "inference_time": "measured",  # Could add timing
153                "message": f"ONNX model loaded successfully. Input: {input_feed[input_name].shape}, Output: {output.shape}",
154            }
155
156        except ImportError:
157            return {"success": False, "error": "onnxruntime not available"}
158        except Exception as e:
159            import traceback
160
161            return {
162                "success": False,
163                "error": f"ONNX validation failed: {e}",
164                "traceback": traceback.format_exc(),
165            }

167    def _validate_gguf_model(self, model_path: str, model_type: str) -> Dict[str, Any]:
168        """Validate GGUF model"""

170            model_dir = Path(model_path)
171            gguf_files = list(model_dir.glob("*.gguf"))
172            if not gguf_files:
173                return {"success": False, "error": "No GGUF files found"}
174
175            gguf_file = gguf_files[0]
176
177            # Check GGUF file header
178            with open(gguf_file, "rb") as f:
179                header = f.read(8)
180                if not header.startswith(b"GGUF"):
181                    return {"success": False, "error": "Invalid GGUF file header"}
182
183            # Try to load with llama.cpp if available
184            llama_main = self._find_llama_main()
185            if llama_main:
186                try:
187                    # Test loading with llama.cpp
188                    cmd = [
189                        llama_main,
190                        "-m",
191                        str(gguf_file),
192                        "-n",
193                        "1",
194                        "--no-display-prompt",
195                    ]
196                    result = subprocess.run(cmd, capture_output=True, text=True, timeout=30)
197
198                    if result.returncode == 0:
199                        return {
200                            "success": True,
201                            "model_file": str(gguf_file),
202                            "message": "GGUF model loaded successfully with llama.cpp",
203                        }
204                    else:
205                        return {
206                            "success": False,
207                            "error": f"llama.cpp loading failed: {result.stderr}",
208                        }
209                except subprocess.TimeoutExpired:
210                    return {"success": False, "error": "llama.cpp loading timed out"}
211                except Exception as e:
212                    return {
213                        "success": False,
214                        "error": f"llama.cpp validation failed: {e}",
215                    }
216            else:
217                return {
218                    "success": True,
219                    "model_file": str(gguf_file),
220                    "message": "GGUF file header valid, llama.cpp not available for full validation",
221                }
222
223        except Exception as e:
224            return {"success": False, "error": f"GGUF validation failed: {e}"}

293    def _validate_torchscript_model(self, model_path: str, model_type: str) -> Dict[str, Any]:
294        """Validate TorchScript model"""
295        try:
296            import torch
297
298            model_path = Path(model_path)
299            if model_path.is_file() and model_path.suffix == ".pt":
300                ts_file = model_path
301            else:
302                ts_files = list(model_path.glob("*.pt"))
303                if not ts_files:
304                    return {"success": False, "error": "No TorchScript files found"}
305                ts_file = ts_files[0]
306            model = torch.jit.load(str(ts_file))
307            model.eval()
308            import inspect
309
310            dummy_input = torch.zeros((1, 8), dtype=torch.long)
311            try:
312                sig = inspect.signature(model.forward)
313                params = list(sig.parameters.keys())
314                if "attention_mask" in params:
315                    dummy_mask = torch.ones((1, 8), dtype=torch.long)
316                    with torch.no_grad():
317                        _ = model(dummy_input, dummy_mask)
318                else:
319                    with torch.no_grad():
320                        _ = model(dummy_input)
321                return {
322                    "success": True,
323                    "model_file": str(ts_file),
324                    "message": "TorchScript model loaded and ran successfully",
325                }
326            except (ValueError, TypeError) as e:
327                # Capsule/no signature，降级为加载级验证
328                return {
329                    "success": True,
330                    "model_file": str(ts_file),
331                    "message": "TorchScript JIT Capsule模型无法自动推理，仅做加载级验证。",
332                    "warning": f"{e}",
333                }
334        except Exception as e:
335            return {"success": False, "error": f"TorchScript validation failed: {e}"}
